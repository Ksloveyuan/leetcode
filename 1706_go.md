func findBall(grid [][]int) []int {
    m,n := len(grid), len(grid[0])
    startPoints := make([]int, n, n)
    for i:=0; i<n; i++ {
        startPoints[i] = i
    }

    stopPoints := make(map[int]struct{})

    for i := 0; i < m; i++ {
        for j := 0; j < n; j++{
            if _, ok := stopPoints[j]; ok {
                continue
            }

            pos := startPoints[j]

            if grid[i][pos] == 1 {
                if (pos+1 < n && grid[i][pos+1] == -1) || pos+1 == n {
                    stopPoints[j] = struct{}{}
                } else {
                    startPoints[j] = pos + 1
                }
            }

            if grid[i][pos] == -1 {
                if (pos-1 >=0 && grid[i][pos-1] == 1) || pos == 0{
                    stopPoints[j] = struct{}{}
                } else {
                    startPoints[j] = pos - 1
                }
            }
        }
    }

    var ans []int
    for i:=0; i<n; i++ {
        if _, ok := stopPoints[i]; ok {
            ans = append(ans, -1)
        } else {
            ans = append(ans, startPoints[i])
        }
    }

    return ans
}
